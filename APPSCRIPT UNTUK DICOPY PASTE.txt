const SHEET_USER = "user";

function json(data){
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}

function doGet(e){
  const mode = e.parameter.mode;
  if(mode === "login") return login(e);
  if(mode === "list") return getData(e);
  return json({status:"invalid"});
}

function login(e){
  const user = (e.parameter.user || "").trim().toLowerCase();
  const pass = (e.parameter.pass || "").trim();
  const sh = SpreadsheetApp.getActive().getSheetByName(SHEET_USER);
  const data = sh.getDataRange().getValues();
  
  for(let i=1; i<data.length; i++){
    if(data[i][1].toString().toLowerCase().trim() == user && data[i][2].toString().trim() == pass){
      return json({ status:"ok", wewenang: data[i][0].toString().trim() });
    }
  }
  return json({status:"fail"});
}

function getData(e){
  const tahun = e.parameter.tahun;
  const wewenang = e.parameter.wewenang;
  const sh = SpreadsheetApp.getActive().getSheetByName(tahun);
  if(!sh) return json({status:"error", message:"Sheet tidak ada"});

  const data = sh.getDataRange().getValues();
  const header = data.shift();
  
  const filteredData = data.filter(r => {
    if (!wewenang || wewenang === "Semua") return true;
    return r[1].toString().trim().toLowerCase() === wewenang.toLowerCase();
  });

  const result = filteredData.map(r => {
    let obj = {};
    header.forEach((h,i) => obj[h] = r[i]);
    return obj;
  });
  
  return json({status:"ok", data: result});
}
